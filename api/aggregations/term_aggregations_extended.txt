#using term aggregations, to get the total quantity purchased by Male customer


GET my-index/_search
{
  "size": 0,
  "aggs": {
    "my_filter": {
      "filter": {
        "term": {
          "customer_gender": "MALE"
        }
      },
      "aggs": {
        "my_filter1": {
          "value_count": {
            "field": "total_quantity"
          }
        }
      }
    }
  }
}
